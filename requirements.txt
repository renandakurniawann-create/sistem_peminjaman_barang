# Requirements instalasi Sistem Peminjaman Barang Bengkel
# Project ini menggunakan React + Vite, jadi dependency utama di-install dengan npm dari package.json, bukan dengan pip.

Node.js >= 20
npm >= 10

# Install dependency project
npm install

# Jalankan development server
npm run dev

# Build untuk production/Vercel
npm run build

# Environment variable wajib
VITE_SUPABASE_URL
VITE_SUPABASE_ANON_KEY
